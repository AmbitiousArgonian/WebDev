// Session data for users.
model Session
{
  id         String   @id @default(cuid())
  sid        String   @unique
  data       String
  expiresAt  DateTime
 // createdAt  DateTime @default(now())
 // updatedAt  DateTime @updatedAt
}

// User model representing application users.
model User 
{   
  id        Int   @id @default(autoincrement())
  scanItems ScanItem[]
}

// ScanItems model representing items scanned by users.
model ScanItem
{
  id                  Int        @id @default(autoincrement())
  name                String     @default("UNKNOWN")
  ean                 String     @default("UNKNOWN")
  ecoScore            Int        @default(-1)
  ecoScoreCategory    String     @default("UNKNOWN")
  nutriScore          Int        @default(-1)
  nutriScoreCategory  String     @default("UNKNOWN")
  content             String     @default("UNKNOWN")
  createdAt           DateTime   @default(now())
  updatedAt           DateTime   @updatedAt
  userId              Int
  user                User?      @relation(fields: [userId], references: [id])
  nutrition           Nutrition[]
}

// Nutrition model storing nutritional information for scanned items.
model Nutrition
{
  id            Int       @id @default(autoincrement())
  value         Int       @default(-1)
  name          String    @default("UNKNOWN")
  createdAt     DateTime  @default(now())
  updatedAt     DateTime  @updatedAt
  scanItemId    Int
  scanItem      ScanItem? @relation(fields: [scanItemId], references: [id])
}
